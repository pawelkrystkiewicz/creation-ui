## Usage

```ts copy
import { Autocomplete } from '@creation-ui/autocomplete'
```

```ts copy

const Orange = () => <div className='bg-red-200 p-3'>Orange</div>

const options = [
  { value: 'apple', label: 'Apple' },
  { value: 'banana', label: 'Banana' },
  { value: <Orange/>, label: 'Orange' },
]

/*
* label can be ReactNode
* that means you can provide your own component as label
* for example to render user with Avatar
*/

const handleChange = (option) => {
  doSomething(option.value)
}

<Autocomplete options={options} onChange={handleChange}/>


```
